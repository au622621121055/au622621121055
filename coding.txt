import RPi.GPIO as GPIO
import time

# GPIO pins for the sensor
TRIG_PIN = 23
ECHO_PIN = 24

# Set the GPIO mode and pins
GPIO.setmode(GPIO.BCM)
GPIO.setup(TRIG_PIN, GPIO.OUT)
GPIO.setup(ECHO_PIN, GPIO.IN)

def measure_distance():
    # Trigger the sensor to send ultrasonic waves
    GPIO.output(TRIG_PIN, True)
    time.sleep(0.00001)
    GPIO.output(TRIG_PIN, False)
    
    # Measure the time it takes for the waves to bounce back
    while GPIO.input(ECHO_PIN) == 0:
        pulse_start = time.time()
    
    while GPIO.input(ECHO_PIN) == 1:
        pulse_end = time.time()
    
    # Calculate distance from time
    pulse_duration = pulse_end - pulse_start
    distance = pulse_duration * 17150  # Speed of sound (343 m/s)
    distance = round(distance, 2)
    
    return distance

try:
    while True:
        distance = measure_distance()
        print(f"Water level: {distance} cm")
        
        # Add logic to send data to IoT platform or take action as needed
        
        time.sleep(1)  # Wait for 1 second before the next reading

except KeyboardInterrupt:
    GPIO.cleanup()  # Cleanup GPIO settings on Ctrl+C exit
